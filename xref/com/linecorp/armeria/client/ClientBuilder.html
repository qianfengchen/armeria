16  package com.linecorp.armeria.client;
17  
18  import static java.util.Objects.requireNonNull;
19  
20  import java.net.URI;
21  
22  import javax.annotation.Nullable;
23  
24  import com.linecorp.armeria.common.Scheme;
25  import com.linecorp.armeria.common.SerializationFormat;
26  import com.linecorp.armeria.common.SessionProtocol;
27  
28  /**
29   * Creates a new client that connects to the specified {@link URI} using the builder pattern. Use the factory
30   * methods in {@link Clients} if you do not have many options to override. If you are creating an
31   * {@link WebClient}, it is recommended to use the {@link WebClientBuilder} or
32   * factory methods in {@link WebClient}.
33   *
34   * <h3>How are decorators and HTTP headers configured?</h3>
35   *
36   * <p>Unlike other options, when a user calls {@link #option(ClientOption, Object)} or {@code options()} with
37   * a {@link ClientOption#DECORATION} or a {@link ClientOption#HTTP_HEADERS}, this builder will not simply
38   * replace the old option but <em>merge</em> the specified option into the previous option value. For example:
39   * <pre>{@code
40   * ClientOptionsBuilder b = new ClientOptionsBuilder();
41   * b.option(ClientOption.HTTP_HEADERS, headersA);
42   * b.option(ClientOption.HTTP_HEADERS, headersB);
43   * b.option(ClientOption.DECORATION, decorationA);
44   * b.option(ClientOption.DECORATION, decorationB);
45   *
46   * ClientOptions opts = b.build();
47   * HttpHeaders httpHeaders = opts.httpHeaders();
48   * ClientDecoration decorations = opts.decoration();
49   * }</pre>
50   * {@code httpHeaders} will contain all HTTP headers of {@code headersA} and {@code headersB}.
51   * If {@code headersA} and {@code headersB} have the headers with the same name, the duplicate header in
52   * {@code headerB} will replace the one with the same name in {@code headerA}.
53   * Similarly, {@code decorations} will contain all decorators of {@code decorationA} and {@code decorationB},
54   * but there will be no replacement but only addition.
55   */
56  public final class ClientBuilder extends AbstractClientOptionsBuilder<ClientBuilder> {
57  
58      @Nullable
59      private final URI uri;
60      @Nullable
61      private final Endpoint endpoint;
62      @Nullable
63      private final Scheme scheme;
64      @Nullable
65      private final SessionProtocol protocol;
66      @Nullable
67      private String path;
68  
69      private SerializationFormat format = SerializationFormat.NONE;
70  
71      private ClientFactory factory = ClientFactory.ofDefault();
72  
73      /**
74       * Creates a new {@link ClientBuilder} that builds the client that connects to the specified {@code uri}.
75       */
76      public ClientBuilder(String uri) {
77          this(URI.create(requireNonNull(uri, "uri")));
78      }
79  
80      /**
81       * Creates a new {@link ClientBuilder} that builds the client that connects to the specified {@link URI}.
82       */
83      public ClientBuilder(URI uri) {
84          this(requireNonNull(uri, "uri"), null, null, null);
85      }
86  
87      /**
88       * Creates a new {@link ClientBuilder} that builds the client that connects to the specified
89       * {@link Endpoint} with the {@code scheme}.
90       */
91      public ClientBuilder(String scheme, Endpoint endpoint) {
92          this(Scheme.parse(requireNonNull(scheme, "scheme")), requireNonNull(endpoint, "endpoint"));
93      }
94  
95      /**
96       * Creates a new {@link ClientBuilder} that builds the client that connects to the specified
97       * {@link Endpoint} with the {@link Scheme}.
98       */
99      public ClientBuilder(Scheme scheme, Endpoint endpoint) {
100         this(null, requireNonNull(scheme, "scheme"), null, requireNonNull(endpoint, "endpoint"));
101     }
102 
103     /**
104      * Creates a new {@link ClientBuilder} that builds the client that connects to the specified
105      * {@link Endpoint} with the {@link SessionProtocol}.
106      */
107     public ClientBuilder(SessionProtocol protocol, Endpoint endpoint) {
108         this(null, null, requireNonNull(protocol, "protocol"), requireNonNull(endpoint, "endpoint"));
109     }
110 
111     private ClientBuilder(@Nullable URI uri, @Nullable Scheme scheme, @Nullable SessionProtocol protocol,
112                           @Nullable Endpoint endpoint) {
113         this.uri = uri;
114         this.scheme = scheme;
115         this.protocol = protocol;
116         this.endpoint = endpoint;
117     }
118 
119     /**
120      * Sets the {@link ClientFactory} of the client. The default is {@link ClientFactory#ofDefault()}.
121      */
122     public ClientBuilder factory(ClientFactory factory) {
123         this.factory = requireNonNull(factory, "factory");
124         return this;
125     }
126 
127     /**
128      * Sets the {@code path} of the client.
129      */
130     public ClientBuilder path(String path) {
131         ensureEndpoint();
132 
133         this.path = requireNonNull(path, "path");
134         return this;
135     }
136 
137     /**
138      * Sets the {@link SerializationFormat} of the client. The default is {@link SerializationFormat#NONE}.
139      */
140     public ClientBuilder serializationFormat(SerializationFormat format) {
141         ensureEndpoint();
142         if (scheme != null) {
143             throw new IllegalStateException("scheme is already given");
144         }
145 
146         this.format = requireNonNull(format, "format");
147         return this;
148     }
149 
150     /**
151      * Returns a newly-created client which implements the specified {@code clientType}, based on the
152      * properties of this builder.
153      *
154      * @throws IllegalArgumentException if the scheme of the {@code uri} specified in
155      *                                  {@link #ClientBuilder(String)} or the specified {@code clientType} is
156      *                                  unsupported for the scheme
157      */
158     public <T> T build(Class<T> clientType) {
159         requireNonNull(clientType, "clientType");
160 
161         if (uri != null) {
162             return factory.newClient(uri, clientType, buildOptions());
163         } else if (path != null) {
164             return factory.newClient(scheme(), endpoint, path, clientType, buildOptions());
165         } else {
166             return factory.newClient(scheme(), endpoint, clientType, buildOptions());
167         }
168     }
169 
170     private Scheme scheme() {
171         return scheme == null ? Scheme.of(format, protocol) : scheme;
172     }
173 
174     private void ensureEndpoint() {
175         if (endpoint == null) {
176             throw new IllegalStateException(
177                     getClass().getSimpleName() + " must be created with an " + Endpoint.class.getSimpleName() +
178                     " to call this method.");
179         }
180     }
181 }
