160     private DefaultServiceRequestContext(
161             ServiceConfig cfg, Channel ch, MeterRegistry meterRegistry, SessionProtocol sessionProtocol,
162             RequestId id, RoutingContext routingContext, RoutingResult routingResult, HttpRequest req,
163             @Nullable SSLSession sslSession, ProxiedAddresses proxiedAddresses, InetAddress clientAddress,
164             boolean requestStartTimeSet, long requestStartTimeNanos,
165             long requestStartTimeMicros) {
166 
167         super(meterRegistry, sessionProtocol, id,
168               requireNonNull(routingContext, "routingContext").method(), routingContext.path(),
169               requireNonNull(routingResult, "routingResult").query(),
170               requireNonNull(req, "req"), null);
171 
172         this.ch = requireNonNull(ch, "ch");
173         this.cfg = requireNonNull(cfg, "cfg");
174         this.routingContext = routingContext;
175         this.routingResult = routingResult;
176         this.sslSession = sslSession;
177         this.proxiedAddresses = requireNonNull(proxiedAddresses, "proxiedAddresses");
178         this.clientAddress = requireNonNull(clientAddress, "clientAddress");
179 
180         log = new DefaultRequestLog(this, cfg.requestContentPreviewerFactory(),
181                                     cfg.responseContentPreviewerFactory());
182         if (requestStartTimeSet) {
183             log.startRequest(ch, sessionProtocol, sslSession, requestStartTimeNanos, requestStartTimeMicros);
184         } else {
185             log.startRequest(ch, sessionProtocol, sslSession);
186         }
187         log.requestHeaders(req.headers());
188 
189         // For the server, request headers are processed well before ServiceRequestContext is created. It means
190         // there is some delay between the actual channel read and this logging, but it's the best we can do for
191         // now.
192         log.requestFirstBytesTransferred();
193 
194         logger = newLogger(cfg);
195 
196         requestTimeoutMillis = cfg.requestTimeoutMillis();
197         maxRequestLength = cfg.maxRequestLength();
198         additionalResponseHeaders = HttpHeaders.of();
199         additionalResponseTrailers = HttpHeaders.of();
200     }

202     private RequestContextAwareLogger newLogger(ServiceConfig cfg) {
203         String loggerName = cfg.loggerName().orElse(null);
204         if (loggerName == null) {
205             loggerName = cfg.route().loggerName();
206         }
207 
208         return new RequestContextAwareLogger(this, LoggerFactory.getLogger(
209                 cfg.server().config().serviceLoggerPrefix() + '.' + loggerName));
210     }

237     @Override
238     public ServiceRequestContext newDerivedContext(RequestId id,
239                                                    @Nullable HttpRequest req,
240                                                    @Nullable RpcRequest rpcReq) {
241         requireNonNull(req, "req");
242         if (rpcRequest() != null) {
243             requireNonNull(rpcReq, "rpcReq");
244         }
245 
246         final DefaultServiceRequestContext ctx = new DefaultServiceRequestContext(
247                 cfg, ch, meterRegistry(), sessionProtocol(), id, routingContext,
248                 routingResult, req, sslSession(), proxiedAddresses(), clientAddress());
249 
250         if (rpcReq != null) {
251             ctx.updateRpcRequest(rpcReq);
252         }
253 
254         final HttpHeaders additionalHeaders = additionalResponseHeaders();
255         if (!additionalHeaders.isEmpty()) {
256             ctx.setAdditionalResponseHeaders(additionalHeaders);
257         }
258 
259         final HttpHeaders additionalTrailers = additionalResponseTrailers();
260         if (!additionalTrailers.isEmpty()) {
261             ctx.setAdditionalResponseTrailers(additionalTrailers);
262         }
263 
264         for (final Iterator<Attribute<?>> i = attrs(); i.hasNext();/* noop */) {
265             ctx.addAttr(i.next());
266         }
267         return ctx;
268     }

270     @SuppressWarnings("unchecked")
271     private <T> void addAttr(Attribute<?> attribute) {
272         final Attribute<T> a = (Attribute<T>) attribute;
273         attr(a.key()).set(a.get());
274     }

357     @Override
358     public void setRequestTimeoutMillis(long requestTimeoutMillis) {
359         if (requestTimeoutMillis < 0) {
360             throw new IllegalArgumentException(
361                     "requestTimeoutMillis: " + requestTimeoutMillis + " (expected: >= 0)");
362         }
363         if (this.requestTimeoutMillis != requestTimeoutMillis) {
364             this.requestTimeoutMillis = requestTimeoutMillis;
365             final RequestTimeoutChangeListener listener = requestTimeoutChangeListener;
366             if (listener != null) {
367                 if (ch.eventLoop().inEventLoop()) {
368                     listener.onRequestTimeoutChange(requestTimeoutMillis);
369                 } else {
370                     ch.eventLoop().execute(() -> listener.onRequestTimeoutChange(requestTimeoutMillis));
371                 }
372             }
373         }
374     }

460     private void updateAdditionalResponseHeaders(
461             AtomicReferenceFieldUpdater<DefaultServiceRequestContext, HttpHeaders> atomicUpdater,
462             Function<HttpHeadersBuilder, HttpHeadersBuilder> valueUpdater) {
463         for (;;) {
464             final HttpHeaders oldValue = atomicUpdater.get(this);
465             final HttpHeaders newValue = valueUpdater.apply(oldValue.toBuilder()).build();
466             if (atomicUpdater.compareAndSet(this, oldValue, newValue)) {
467                 return;
468             }
469         }
470     }

474         return removeAdditionalResponseHeader(additionalResponseHeadersUpdater, name);
475     }

477     private boolean removeAdditionalResponseHeader(
478             AtomicReferenceFieldUpdater<DefaultServiceRequestContext, HttpHeaders> atomicUpdater,
479             CharSequence name) {
480         requireNonNull(name, "name");
481         for (;;) {
482             final HttpHeaders oldValue = atomicUpdater.get(this);
483             if (oldValue.isEmpty() || !oldValue.contains(name)) {
484                 return false;
485             }
486 
487             final HttpHeaders newValue = oldValue.toBuilder().removeAndThen(name).build();
488             if (atomicUpdater.compareAndSet(this, oldValue, newValue)) {
489                 return true;
490             }
491         }
492     }

554     /**
555      * Sets the listener that is notified when the {@linkplain #requestTimeoutMillis()} request timeout} of

569     @Override
570     public String toString() {
571         String strVal = this.strVal;
572         if (strVal != null) {
573             return strVal;
574         }
575 
576         final StringBuilder buf = new StringBuilder(96);
577 
578         // Prepend the current channel information if available.
579         final Channel ch = channel();
580         final boolean hasChannel = ch != null;
581         if (hasChannel) {
582             buf.append(ch);
583 
584             final InetAddress remote = ((InetSocketAddress) remoteAddress()).getAddress();
585             final InetAddress client = clientAddress();
586             if (remote != null && !remote.equals(client)) {
587                 buf.append("[C:").append(client.getHostAddress()).append(']');
588             }
589         }
590 
591         buf.append('[')
592            .append(sessionProtocol().uriText())
593            .append("://")
594            .append(virtualHost().defaultHostname());
595 
596         final InetSocketAddress laddr = localAddress();
597         if (laddr != null) {
598             buf.append(':').append(laddr.getPort());
599         } else {
600             buf.append(":-1"); // Port unknown.
601         }
602 
603         buf.append(path())
604            .append('#')
605            .append(method())
606            .append(']');
607 
608         strVal = buf.toString();
609         if (hasChannel) {
610             this.strVal = strVal;
611         }
612         return strVal;
613     }
