81      /**
82       * Validate {@code sslContext} is configured properly. If {@code sslContext} is configured as client
83       * context, or key store password is not given to key store when {@code sslContext} is created using key
84       * manager factory, the validation will fail and an {@link SSLException} will be raised.
85       */
86      private static SslContext validateSslContext(SslContext sslContext) throws SSLException {
87          if (!sslContext.isServer()) {
88              throw new IllegalArgumentException("sslContext: " + sslContext + " (expected: server context)");
89          }
90  
91          SSLEngine serverEngine = null;
92          SSLEngine clientEngine = null;
93  
94          try {
95              serverEngine = sslContext.newEngine(ByteBufAllocator.DEFAULT);
96              serverEngine.setUseClientMode(false);
97              serverEngine.setNeedClientAuth(false);
98  
99              final SslContext sslContextClient =
100                     buildSslContext(SslContextBuilder::forClient, sslContextBuilder -> {});
101             clientEngine = sslContextClient.newEngine(ByteBufAllocator.DEFAULT);
102             clientEngine.setUseClientMode(true);
103 
104             final ByteBuffer appBuf = ByteBuffer.allocate(clientEngine.getSession().getApplicationBufferSize());
105             final ByteBuffer packetBuf = ByteBuffer.allocate(clientEngine.getSession().getPacketBufferSize());
106 
107             clientEngine.wrap(appBuf, packetBuf);
108             appBuf.clear();
109             packetBuf.flip();
110             serverEngine.unwrap(packetBuf, appBuf);
111         } catch (SSLException e) {
112             throw new SSLException("failed to validate SSL/TLS configuration: " + e.getMessage(), e);
113         } finally {
114             ReferenceCountUtil.release(serverEngine);
115             ReferenceCountUtil.release(clientEngine);
116         }
117 
118         return sslContext;
119     }

233     /**
234      * Configures SSL or TLS of this {@link VirtualHost} with the specified {@code keyCertChainFile},
235      * {@code keyFile}, {@code keyPassword} and {@code tlsCustomizer}.
236      */
237     public VirtualHostBuilder tls(File keyCertChainFile, File keyFile, @Nullable String keyPassword,
238                                   Consumer<SslContextBuilder> tlsCustomizer) throws SSLException {
239         requireNonNull(keyCertChainFile, "keyCertChainFile");
240         requireNonNull(keyFile, "keyFile");
241         requireNonNull(tlsCustomizer, "tlsCustomizer");
242 
243         if (!keyCertChainFile.exists()) {
244             throw new SSLException("non-existent certificate chain file: " + keyCertChainFile);
245         }
246         if (!keyCertChainFile.canRead()) {
247             throw new SSLException("cannot read certificate chain file: " + keyCertChainFile);
248         }
249         if (!keyFile.exists()) {
250             throw new SSLException("non-existent key file: " + keyFile);
251         }
252         if (!keyFile.canRead()) {
253             throw new SSLException("cannot read key file: " + keyFile);
254         }
255 
256         tls(buildSslContext(() -> SslContextBuilder.forServer(keyCertChainFile, keyFile, keyPassword),
257                             tlsCustomizer));
258         return this;
259     }

274     private static SslContext buildSslContext(
275             Supplier<SslContextBuilder> builderSupplier,
276             Consumer<SslContextBuilder> tlsCustomizer) throws SSLException {
277         try {
278             return BouncyCastleKeyFactoryProvider.call(() -> SslContextUtil.createSslContext(
279                     builderSupplier, false, tlsCustomizer));
280         } catch (RuntimeException | SSLException e) {
281             throw e;
282         } catch (Exception e) {
283             throw new SSLException("failed to configure TLS: " + e, e);
284         }
285     }

374     /**
375      * Decorates and binds the specified {@link HttpServiceWithRoutes} at multiple {@link Route}s
376      * of the default {@link VirtualHost}.
377      *
378      * @param serviceWithRoutes the {@link HttpServiceWithRoutes}.
379      * @param decorators the decorator functions, which will be applied in the order specified.
380      */
381     public VirtualHostBuilder service(
382             HttpServiceWithRoutes serviceWithRoutes,
383             Iterable<Function<? super HttpService, ? extends HttpService>> decorators) {
384         requireNonNull(serviceWithRoutes, "serviceWithRoutes");
385         requireNonNull(serviceWithRoutes.routes(), "serviceWithRoutes.routes()");
386         requireNonNull(decorators, "decorators");
387 
388         HttpService decorated = serviceWithRoutes;
389         for (Function<? super HttpService, ? extends HttpService> d : decorators) {
390             checkNotNull(d, "decorators contains null: %s", decorators);
391             decorated = d.apply(decorated);
392             checkNotNull(decorated, "A decorator returned null: %s", d);
393         }
394 
395         final HttpService finalDecorated = decorated;
396         serviceWithRoutes.routes().forEach(route -> service(route, finalDecorated));
397         return this;
398     }

579     @Nullable
580     private Function<? super HttpService, ? extends HttpService> getRouteDecoratingService(
581             @Nullable VirtualHostBuilder defaultVirtualHostBuilder) {
582         final List<RouteDecoratingService> routeDecoratingServices;
583         if (defaultVirtualHostBuilder != null) {
584             routeDecoratingServices = ImmutableList.<RouteDecoratingService>builder()
585                     .addAll(this.routeDecoratingServices)
586                     .addAll(defaultVirtualHostBuilder.routeDecoratingServices)
587                     .build();
588         } else {
589             routeDecoratingServices = ImmutableList.copyOf(this.routeDecoratingServices);
590         }
591 
592         if (!routeDecoratingServices.isEmpty()) {
593             return RouteDecoratingService.newDecorator(
594                     Routers.ofRouteDecoratingService(routeDecoratingServices));
595         } else {
596             return null;
597         }
598     }

842     /**
843      * Returns a newly-created {@link VirtualHost} based on the properties of this builder and the services
844      * added to this builder.
845      */
846     VirtualHost build(VirtualHostBuilder template) {
847         requireNonNull(template, "template");
848 
849         if (defaultHostname == null) {
850             if ("*".equals(hostnamePattern)) {
851                 defaultHostname = SystemInfo.hostname();
852             } else if (hostnamePattern.startsWith("*.")) {
853                 defaultHostname = hostnamePattern.substring(2);
854             } else {
855                 defaultHostname = hostnamePattern;
856             }
857         } else {
858             ensureHostnamePatternMatchesDefaultHostname(hostnamePattern, defaultHostname);
859         }
860 
861         // Retrieve all settings as a local copy. Use default builder's properties if not set.
862         final long requestTimeoutMillis =
863                 this.requestTimeoutMillis != null ?
864                 this.requestTimeoutMillis : template.requestTimeoutMillis;
865         final long maxRequestLength =
866                 this.maxRequestLength != null ?
867                 this.maxRequestLength : template.maxRequestLength;
868         final boolean verboseResponses =
869                 this.verboseResponses != null ?
870                 this.verboseResponses : template.verboseResponses;
871         final ContentPreviewerFactory requestContentPreviewerFactory =
872                 this.requestContentPreviewerFactory != null ?
873                 this.requestContentPreviewerFactory : template.requestContentPreviewerFactory;
874         final ContentPreviewerFactory responseContentPreviewerFactory =
875                 this.responseContentPreviewerFactory != null ?
876                 this.responseContentPreviewerFactory : template.responseContentPreviewerFactory;
877         final RejectedRouteHandler rejectedRouteHandler =
878                 this.rejectedRouteHandler != null ?
879                 this.rejectedRouteHandler : template.rejectedRouteHandler;
880 
881         final AccessLogWriter accessLogWriter;
882         final boolean shutdownAccessLogWriterOnStop;
883         if (this.accessLogWriter != null) {
884             accessLogWriter = this.accessLogWriter;
885             shutdownAccessLogWriterOnStop = this.shutdownAccessLogWriterOnStop;
886         } else {
887             accessLogWriter = template.accessLogWriter;
888             shutdownAccessLogWriterOnStop = template.shutdownAccessLogWriterOnStop;
889         }
890 
891         final Function<VirtualHost, Logger> accessLoggerMapper =
892                 this.accessLoggerMapper != null ?
893                 this.accessLoggerMapper : template.accessLoggerMapper;
894 
895         assert requestContentPreviewerFactory != null;
896         assert responseContentPreviewerFactory != null;
897         assert rejectedRouteHandler != null;
898         assert accessLogWriter != null;
899         assert accessLoggerMapper != null;
900 
901         final List<ServiceConfigBuilder> serviceConfigBuilders =
902                 getServiceConfigBuilders(template);
903         final List<ServiceConfig> serviceConfigs = serviceConfigBuilders.stream().map(cfgBuilder -> {
904             return cfgBuilder.build(requestTimeoutMillis, maxRequestLength, verboseResponses,
905                                     requestContentPreviewerFactory, responseContentPreviewerFactory,
906                                     accessLogWriter, shutdownAccessLogWriterOnStop);
907         }).collect(toImmutableList());
908 
909         final ServiceConfig fallbackServiceConfig =
910                 new ServiceConfigBuilder(Route.ofCatchAll(), FallbackService.INSTANCE)
911                         .build(requestTimeoutMillis, maxRequestLength, verboseResponses,
912                                requestContentPreviewerFactory, responseContentPreviewerFactory,
913                                accessLogWriter, shutdownAccessLogWriterOnStop);
914 
915         SslContext sslContext = this.sslContext != null ? this.sslContext : template.sslContext;
916         final boolean tlsSelfSigned = this.tlsSelfSigned != null ? this.tlsSelfSigned : template.tlsSelfSigned;
917         if (sslContext == null && tlsSelfSigned) {
918             try {
919                 final SelfSignedCertificate ssc = new SelfSignedCertificate(defaultHostname);
920                 tls(ssc.certificate(), ssc.privateKey());
921                 sslContext = this.sslContext;
922             } catch (Exception e) {
923                 throw new RuntimeException("failed to create a self signed certificate", e);
924             }
925         }
926 
927         final VirtualHost virtualHost =
928                 new VirtualHost(defaultHostname, hostnamePattern, sslContext,
929                                 serviceConfigs, fallbackServiceConfig, rejectedRouteHandler,
930                                 accessLoggerMapper, requestTimeoutMillis, maxRequestLength,
931                                 verboseResponses, requestContentPreviewerFactory,
932                                 responseContentPreviewerFactory, accessLogWriter,
933                                 shutdownAccessLogWriterOnStop);
934 
935         final Function<? super HttpService, ? extends HttpService> decorator =
936                 getRouteDecoratingService(template);
937         return decorator != null ? virtualHost.decorate(decorator) : virtualHost;
938     }
